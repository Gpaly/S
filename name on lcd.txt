// Include the library code for the LiquidCrystal
#include <LiquidCrystal.h>

// Create an LCD object. Parameters: (rs, enable, d4, d5, d6, d7)
LiquidCrystal lcd(12, 11, 5, 4, 3, 2);

void setup() {
  // Set up the LCD's number of columns and rows (16x2 LCD)
  lcd.begin(16, 2);

  // Clear the LCD screen
  lcd.clear();
}

void loop() {
  // Print a message on the first row
  lcd.print(" Hello world!");

  // Set the cursor to column 0, line 1 (second row)
  lcd.setCursor(0, 1);

  // Print a message on the second row
  lcd.print(" LCD Tutorial");

  // Keep the message displayed
  while (true); // Infinite loop to keep the message on the screen
}
